    test_bulk_create_returns_201_with_spool_ids:
      POST to "/api/v1/filament-types/bulk-create" with _bulk_payload(test_material_type, quantity=3), headers=auth_headers
      assert response.status_code == 201
      data = response.json()
      assert "filament_type_id" in data

    test_bulk_create_quantity_1_creates_one_spool:
      POST with quantity=1
      assert status_code == 201
      assert len(data["spool_ids"]) == 1

    test_bulk_create_spool_id_format:
      import re at top of file if not already present
      POST with quantity=2
      assert status_code == 201
      for spool_id in data["spool_ids"]:
          assert re.match(r"^FIL-\d{3,}$", spool_id), f"Spool ID {spool_id!r} does not match FIL-NNN format"

    test_bulk_create_spools_are_unlabeled:
      POST with quantity=2
      assert status_code == 201
      # Verify via the spools sub-resource: GET /api/v1/filament-types/{filament_type_id}/spools
      ft_id = data["filament_type_id"]
      spools_response = await client.get(f"/api/v1/filament-types/{ft_id}/spools", headers=auth_headers)
      assert spools_response.status_code == 200
      spools = spools_response.json()
      assert len(spools) == 2
      assert all(s["is_labeled"] is False for s in spools)

    test_bulk_create_requires_auth:
      POST to bulk-create using unauthenticated_client (no auth_headers)
      assert response.status_code in [401, 403]

    test_bulk_create_quantity_zero_rejected:
      POST with quantity=0 (add quantity field to _bulk_payload call or build payload inline)
      assert response.status_code == 422

    test_bulk_create_quantity_over_limit_rejected:
      POST with quantity=21
      assert response.status_code == 422

    test_bulk_create_finds_existing_filament_type:
      POST twice with identical brand="Duplicate Brand", color="Same Color", material_type_id=str(test_material_type.id)
      first_id = first_response.json()["filament_type_id"]
      second_id = second_response.json()["filament_type_id"]
      assert first_id == second_id, "Duplicate FilamentType should be reused, not re-created"
      # Also assert total spools: GET /filament-types/{id}/spools should show 6 spools (3+3)
      spools_response = await client.get(f"/api/v1/filament-types/{first_id}/spools", headers=auth_headers)
      assert len(spools_response.json()) == 6
